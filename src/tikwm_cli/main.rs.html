1use anyhow::Context;
2use anyhow::ensure;
3use std::path::PathBuf;
4use url::Url;
5
6#[derive(argh::FromArgs)]
7#[argh(description = "A small CLI to download tiktok videos")]
8struct Options {
9    #[argh(positional, description = "the post url")]
10    pub post: String,
11
12    #[argh(
13        option,
14        short = 'o',
15        default = "PathBuf::from(\".\")",
16        description = "the output directory"
17    )]
18    pub out_dir: PathBuf,
19}
20
21fn main() -> anyhow::Result<()> {
22    let options: Options = argh::from_env();
23    let tokio_rt = tokio::runtime::Builder::new_multi_thread()
24        .enable_all()
25        .build()?;
26
27    tokio_rt.block_on(async_main(options))?;
28
29    Ok(())
30}
31
32async fn async_main(options: Options) -> anyhow::Result<()> {
33    let client = tikwm::Client::new();
34
35    let video_id: u64 = match Url::parse(&options.post) {
36        Ok(post_url) => post_url
37            .path_segments()
38            .context("post url missing path")?
39            .next_back()
40            .context("missing video id")?
41            .parse()?,
42        Err(_error) => options.post.parse()?,
43    };
44
45    let file_name = format!("{video_id}.mp4");
46    let out_path = options.out_dir.join(file_name);
47
48    if tokio::fs::try_exists(&out_path).await? {
49        eprintln!("file exists, skipping...");
50        return Ok(());
51    }
52
53    eprintln!("creating video download...");
54    let create_download_response = client
55        .create_download_task(&options.post)
56        .await
57        .context("failed to create video download")?;
58    ensure!(create_download_response.id == video_id);
59
60    eprintln!("getting download task result...");
61    let download_task_result = client
62        .get_task_result(&create_download_response.task_id)
63        .await
64        .context("failed to create video download")?;
65
66    eprintln!(
67        "downloading video from \"{}\"",
68        download_task_result.download_url.as_str()
69    );
70    nd_util::download_to_path(
71        &client.client,
72        download_task_result.download_url.as_str(),
73        out_path,
74    )
75    .await?;
76
77    Ok(())
78}
