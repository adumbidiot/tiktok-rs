1use anyhow::ensure;
2use anyhow::Context;
3use std::path::PathBuf;
4use tiktok::Url;
5
6#[derive(argh::FromArgs)]
7#[argh(description = "A small CLI to download tiktok videos")]
8struct Options {
9    #[argh(positional, description = "the post url")]
10    pub post: String,
11
12    #[argh(
13        option,
14        short = 'o',
15        default = "PathBuf::from(\".\")",
16        description = "the output directory"
17    )]
18    pub out_dir: PathBuf,
19}
20
21fn main() -> anyhow::Result<()> {
22    let options: Options = argh::from_env();
23    let tokio_rt = tokio::runtime::Builder::new_multi_thread()
24        .enable_all()
25        .build()?;
26
27    tokio_rt.block_on(async_main(options))?;
28
29    Ok(())
30}
31
32async fn async_main(options: Options) -> anyhow::Result<()> {
33    let client = tiktok::Client::new();
34
35    let video_id = match Url::parse(&options.post) {
36        Ok(post_url) => post_url
37            .path_segments()
38            .context("post url missing path")?
39            .next_back()
40            .context("missing video id")?
41            .parse()?,
42        Err(_error) => options.post.parse()?,
43    };
44
45    let file_name = format!("{video_id}.mp4");
46    let out_path = options.out_dir.join(file_name);
47
48    if tokio::fs::try_exists(&out_path).await? {
49        eprintln!("file exists, skipping...");
50        return Ok(());
51    }
52
53    eprintln!("fetching feed...");
54    let feed_cursor = client
55        .get_feed(Some(video_id))
56        .await
57        .context("failed to get post")?;
58    let post = feed_cursor.aweme_list.first().context("missing post")?;
59    ensure!(post.aweme_id == video_id);
60
61    let video_url = &post
62        .video
63        .download_addr
64        .url_list
65        .first()
66        .context("missing download url")?;
67
68    eprintln!("downloading video from \"{}\"", video_url.as_str());
69
70    nd_util::download_to_path(&client.client, video_url.as_str(), out_path).await?;
71
72    Ok(())
73}
